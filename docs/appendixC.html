C Generating fake data

Hereâ€™s the code I used to create the respondent information table

library(charlatan) # library of fake data
library(glue) # library for pasting together variables

email_domains <- c("@gmail.com", "@hotmail.com", "@outlook.com", "@me.com", "@yahoo.com")

respondent_data <- ch_generate('name', 'job', 'phone_number', n = 502) %>% 
                    separate(name, "first", extra = "drop", remove=FALSE) %>% 
                    mutate(email = glue("{first_lower}{email_domain}", 
                                          first_lower = tolower(first), 
                                          email_domain = sample(email_domains, 502, replace=TRUE) ), 
                           gender = sample(c("male", "female", "other", "prefer not to say"), 502, replace=TRUE, prob=c(0.55, 0.35, 0.05, 0.05)), 
                           age = sample(c("Under 18", "18-34", "35-54", "55+"), 502, replace=TRUE, prob = c(0.05, 0.25, 0.40, 0.30))
        )

write_csv(respondent_data, "./sample_data/respondent_data.csv")
